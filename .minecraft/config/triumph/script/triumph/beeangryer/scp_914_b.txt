//Set icon item
setIcon(<modularmachinery:blockcontroller>)

//Set title String
setTitle("SCP-914-B, Assembled")

//Set description String
setDescription("Right click the controller of SCP-914-B while it's set to any of the correct modes.")

//Add a parent advancement. The syntax is "modid:folder/path/to/the/advancement/then/filename"
addParent("triumph:beeangryer/root")

//Set the position
setPos(0,0)

//Hide the connection lines, so I can have a pretty shape
hideLines()

//Adds criteria named "clickedBlock" with the trigger type "triumph:right_click_block". This function returns the criteria as an object
criteria = addCriteria("clickedBlock", "triumph:right_click_block")

//Sets the required block for the criteria. This is the block that needs to be right clicked by the player.
criteria.setBlock(<modularmachinery:blockcontroller, tile:{machine:"modularmachinery:scp_914_b_1", status:1}>)
criteria.setBlock(<modularmachinery:blockcontroller, tile:{machine:"modularmachinery:scp_914_b_3", status:1}>)
